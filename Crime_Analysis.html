#create clusters from 2 to 20 and find out using the Elbow curve which Value of K makes most sense
X = data
distorsions = []
for k in range(2, 20):
    k_means = KMeans(n_clusters=k)
    k_means.fit(X)
    distorsions.append(k_means.inertia_)
 
fig = plt.figure(figsize=(15, 8))
plt.plot(range(2, 20), distorsions)
plt.grid(True)
plt.title('Elbow curve')
#Looking at this k=5 make the most sense to me.
